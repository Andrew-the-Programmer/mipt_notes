$\Align{
\Cases{
\dot{x} = x^2 + z^2 \\
\dot{y} = 2(xy - xz^3) \\
\dot{z} = 2xz
} \\
(1)\ и\ (2) \\
\frac{\d x}{x^2 + z^2} = \frac{\d y}{2(xy - xz^3)} = \frac{\d z}{2xz} \\
\frac{\d (x+z)}{(x+z)^2} = \frac{\d (x-z)}{(x-z)^2} \\
\frac{1}{x + z} = \frac{1}{x - z} + C \\
u_1 = \frac{z}{x^2 + z^2} \\
(2)\ и\ (3) \\
\frac{\d y}{y - z^3} = \frac{\d z}{z} = \frac{\d (z^3)}{3 z^3} \\
u_2 = \frac{z}{2y + z^3} \\
\textit{Общее решение:} \\
U = F(u_1, u_2) - ПИ \\
F \in C^1 \\
\textit{Решение ЗК:} \\
\Cases{
u_1 = \frac{z}{x^2 + z^2} \\
u_2 = \frac{z}{2y + z^3} \\
x^2 - z^2 = 2
}
}$
